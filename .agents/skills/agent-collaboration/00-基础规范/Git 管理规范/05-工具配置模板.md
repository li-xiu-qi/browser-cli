## Git 配置模板

### 全局配置（~/.gitconfig）

```ini
[user]
    name = 你的名字
    email = 你的邮箱

[core]
    # 自动处理行尾符
    autocrlf = false
    # 忽略文件权限变化
    filemode = false
    # 编码
    quotepath = false

[init]
    defaultBranch = master

[push]
    default = simple

[pull]
    rebase = false

[log]
    # 日期格式
    date = format:%Y-%m-%d %H:%M

[format]
    # 提交信息格式
    pretty = format:%h %ad | %s%d [%an]
```

### 仓库本地配置

在知识库目录下执行：

```bash
# 配置用户名（针对当前仓库）
git config user.name "你的名字"
git config user.email "你的邮箱"

# 配置提交模板
git config commit.template .gitmessage

# 查看配置
git config --list --local
```

## Git 别名配置

### 添加到 ~/.gitconfig

```ini
[alias]
    # 状态查看
    st = status
    sts = status -s

    # 提交
    cm = commit -m
    cam = commit -am
    amend = commit --amend

    # 日志查看
    lg = log --oneline --graph --all
    lga = log --oneline --all
    lgd = log --oneline --since="1 day ago"
    lgw = log --oneline --since="1 week ago"

    # 分支
    br = branch
    co = checkout
    cob = checkout -b

    # 差异查看
    df = diff
    dfs = diff --staged

    # 撤销
    unstage = reset HEAD
    uncommit = reset --soft HEAD~1

    # 快捷推送
    pushom = push origin master
    pullom = pull origin master

    # 查找
    find = log --all --full-history --
    grep = log --all --grep
```

### 使用示例

```bash
# 使用别名
git st              # git status
git cm "feat: xxx"  # git commit -m "feat: xxx"
git lg              # git log --oneline --graph --all
git lgw             # 查看本周提交
```

## 常用快捷命令脚本

### 保存为 git-aliases.sh

```bash
#!/bin/bash
# Git 快捷命令函数

# 快速提交
gcm() {
    git commit -m "$1"
}

# 添加并提交
gac() {
    git add -A
    git commit -m "$1"
}

# 添加、提交、推送
gacp() {
    git add -A
    git commit -m "$1"
    git push origin master
}

# 查看今日提交
gtoday() {
    git log --oneline --since="today" --author="$(git config user.name)"
}

# 查看本周提交
gweek() {
    git log --oneline --since="1 week ago" --author="$(git config user.name)"
}

# 创建临时分支并切换
gtemp() {
    git checkout -b "temp-$1"
}

# 合并临时分支并删除
gmerge() {
    branch=$(git branch --show-current)
    git checkout master
    git merge "$branch"
    git branch -d "$branch"
}

# 搜索内容在历史中
gsearch() {
    git log -S "$1" --all
}

# 搜索提交信息
ggrep() {
    git log --all --grep="$1"
}
```

### 使用方式

```bash
# 添加到 ~/.bashrc 或 ~/.zshrc
source /path/to/git-aliases.sh

# 使用
gcm "feat: 添加新笔记"
gacp "chore: 更新索引"
gtoday
gsearch "关键词"
```

## 初始化脚本

### init-repo.sh

```bash
#!/bin/bash
# 初始化知识库 Git 仓库

echo "初始化知识库 Git 仓库..."

# 初始化 Git
git init

# 创建 .gitignore
cat > .gitignore << 'EOF'
# Obsidian
.obsidian/workspace.json
.obsidian/workspaces.json
.obsidian/workspace-mobile.json
.obsidian/graph.json
.obsidian/cache/

# System
.DS_Store
Thumbs.db

# Dev
**/__pycache__/
**/*.pyc
**/node_modules/
**/.env
**/*.log

# Large files
**/*.mp4
**/*.mp3
**/*.zip
**/*.pdf

# Personal config
**/settings.local.json
**/mcp.json

# Browser data
**/*_user_data/
**/browser_user_data/

# Resources
resources/参考项目/
EOF

# 创建 .gitmessage
cat > .gitmessage << 'EOF'
类型: 简短描述

# 类型选项:
# feat:     新增内容/功能
# refactor: 重构/结构调整
# chore:    日常维护
# fix:      修正错误
# docs:     文档更新
EOF

# 配置 Git
git config user.name "$(whoami)"
git config commit.template .gitmessage

# 创建目录结构
mkdir -p Areas/_探索中
mkdir -p Areas/_学习区
mkdir -p Projects
mkdir -p MOC
mkdir -p Archives

# 初始提交
git add .
git commit -m "chore: 初始化知识库仓库"

echo "完成！"
```
