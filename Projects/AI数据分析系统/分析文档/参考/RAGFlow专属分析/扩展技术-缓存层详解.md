```python
# common/cache_manager.py

import redis
import json
import pickle
import hashlib
from functools import wraps
from typing import Optional, Any
import time

class CacheManager:
    """多级缓存管理器"""
    
    def __init__(self, redis_hosts, local_maxsize=1000):
        """
        Args:
            redis_hosts: Redis 集群地址列表
            local_maxsize: 本地缓存最大条目数
        """
        # L2: Redis 集群
        self.redis = redis.RedisCluster(
            startup_nodes=[{'host': h, 'port': 6379} for h in redis_hosts],
            decode_responses=True,
            skip_full_coverage_check=True
        )
        
        # L1: 本地 LRU 缓存
        self.local_cache = {}
        self.local_maxsize = local_maxsize
        self.local_access_order = []
    
    def _local_get(self, key: str) -> Optional[Any]:
        """L1: 本地缓存读取"""
        if key in self.local_cache:
            # 更新访问顺序
            self.local_access_order.remove(key)
            self.local_access_order.append(key)
            return self.local_cache[key]
        return None
    
    def _local_set(self, key: str, value: Any):
        """L1: 本地缓存写入"""
        # 淘汰旧数据
        if len(self.local_cache) >= self.local_maxsize:
            oldest = self.local_access_order.pop(0)
            del self.local_cache[oldest]
        
        self.local_cache[key] = value
        self.local_access_order.append(key)
    
    def _local_delete(self, key: str):
        """L1: 本地缓存删除"""
        if key in self.local_cache:
            del self.local_cache[key]
            self.local_access_order.remove(key)
    
    def get(self, key: str, use_local=True) -> Optional[Any]:
        """
        获取缓存
        
        顺序：L1本地 -> L2Redis -> None
        """
        # L1: 本地缓存
        if use_local:
            value = self._local_get(key)
            if value is not None:
                return value
        
        # L2: Redis
        value = self.redis.get(key)
        if value:
            data = pickle.loads(value)
            # 回填本地缓存
            if use_local:
                self._local_set(key, data)
            return data
        
        return None
    
    def set(self, key: str, value: Any, ttl: int = 300, use_local=True):
        """
        设置缓存
        
        Args:
            key: 缓存键
            value: 缓存值（任意可序列化对象）
            ttl: 过期时间（秒），默认5分钟
            use_local: 是否写入本地缓存
        """
        # L1: 本地缓存（短 TTL）
        if use_local:
            self._local_set(key, value)
        
        # L2: Redis
        serialized = pickle.dumps(value)
        self.redis.setex(key, ttl, serialized)
    
    def delete(self, key: str):
        """删除缓存（双删）"""
        # L1: 删本地
        self._local_delete(key)
        
        # L2: 删 Redis
        self.redis.delete(key)
    
    def delete_pattern(self, pattern: str):
        """按模式删除缓存"""
        keys = self.redis.keys(pattern)
        if keys:
            self.redis.delete(*keys)
            # 清理本地缓存
            for key in keys:
                self._local_delete(key)


# 全局缓存实例
cache = CacheManager(
    redis_hosts=['redis-1', 'redis-2', 'redis-3'],
    local_maxsize=1000
)
```

### 4.2 装饰器自动缓存

```python
# common/cache_decorator.py

from common.cache_manager import cache
import hashlib
import json


def cached(ttl: int = 300, key_prefix: str = '', cache_empty: bool = False):
    """
    自动缓存装饰器
    
    Args:
        ttl: 缓存过期时间（秒）
        key_prefix: 缓存键前缀
        cache_empty: 是否缓存空结果
    """
    def decorator(func):
        def wrapper(*args, **kwargs):
            # 生成缓存键
            cache_key = _generate_cache_key(key_prefix, func.__name__, args, kwargs)
            
            # 尝试读取缓存
            result = cache.get(cache_key)
            if result is not None:
                return result
            
            # 执行函数
            result = func(*args, **kwargs)
            
            # 写入缓存（空结果可选缓存）
            if result is not None or cache_empty:
                cache.set(cache_key, result, ttl)
            
            return result
        
        # 附加清除缓存方法
        wrapper.cache_clear = lambda *args, **kwargs: cache.delete(
            _generate_cache_key(key_prefix, func.__name__, args, kwargs)
        )
        
        wrapper.cache_clear_all = lambda: cache.delete_pattern(f'{key_prefix}:*')
        
        return wrapper
    return decorator


def _generate_cache_key(prefix: str, func_name: str, args, kwargs) -> str:
    """生成缓存键"""
    # 序列化参数
    key_data = {
        'func': func_name,
        'args': args,
        'kwargs': kwargs
    }
    key_str = json.dumps(key_data, sort_keys=True, default=str)
    
    # MD5 哈希（避免键过长）
    hash_val = hashlib.md5(key_str.encode()).hexdigest()[:16]
    
    return f'{prefix}:{func_name}:{hash_val}'


def cache_evict(key_prefix: str, pattern: str = ''):
    """
    缓存清除装饰器
    用于写操作后清除相关缓存
    """
    def decorator(func):
        def wrapper(*args, **kwargs):
            result = func(*args, **kwargs)
            
            # 清除缓存
            if pattern:
                cache.delete_pattern(f'{key_prefix}:{pattern}')
            else:
                cache.delete_pattern(f'{key_prefix}:*')
            
            return result
        return wrapper
    return decorator
```

```python
# api/db/services/knowledgebase_service.py

from common.cache_decorator import cached, cache_evict

class KnowledgebaseService:
    
    @classmethod
    @cached(ttl=600, key_prefix='kb')  # 缓存10分钟
    def get_by_id(cls, kb_id: str):
        """获取知识库详情 - 自动缓存"""
        return cls.model.select().where(cls.model.id == kb_id).first()
    
    @classmethod
    @cached(ttl=300, key_prefix='kb_list')  # 缓存5分钟
    def get_list(cls, tenant_id: str, page: int = 1, size: int = 20):
        """获取知识库列表 - 自动缓存"""
        return cls.model.select().where(
            cls.model.tenant_id == tenant_id
        ).paginate(page, size)
    
    @classmethod
    @cache_evict(key_prefix='kb')  # 清除 kb:* 的缓存
    def update(cls, kb_id: str, **kwargs):
        """更新知识库 - 自动清除缓存"""
        result = cls.model.update(**kwargs).where(
            cls.model.id == kb_id
        ).execute()
        
        # 清除相关缓存
        cache.delete(f'kb:get_by_id:{kb_id}')
        cache.delete_pattern(f'kb_list:*')
        
        return result
    
    @classmethod
    @cache_evict(key_prefix='kb')  # 清除所有知识库缓存
    def delete(cls, kb_id: str):
        """删除知识库 - 清除所有相关缓存"""
        return cls.model.delete().where(cls.model.id == kb_id).execute()
```

## 五、缓存问题与解决方案

### 5.1 缓存穿透

```
问题：查询不存在的数据，每次都打到数据库

攻击：GET /user/999999（不存在的用户）
        ↓
    查缓存（miss）
        ↓
    查数据库（miss）
        ↓
    返回 404
    
    循环 10000 次 → 数据库被打爆

解决方案1：缓存空值
    cache.set('user:999999', None, ttl=60)  # 缓存空结果

解决方案2：布隆过滤器
    先查布隆过滤器，如果不存在直接返回，不查缓存和数据库
```

### 5.2 缓存击穿

```
问题：热点 key 过期瞬间，大量请求打到数据库

场景：知识库 KB-123 缓存过期
    1ms 内 1000 个请求同时到达
        ↓
    都发现缓存 miss
        ↓
    都去查数据库
        ↓
    数据库瞬时压力暴增

解决方案1：互斥锁（单飞）
    只有第一个请求查数据库，其他等待
    
解决方案2：逻辑过期
    不设置 TTL，而是设置逻辑过期时间
    过期后异步更新，旧数据继续服务
    
解决方案3：随机 TTL
    TTL = 300 + random(0, 60)  # 分散过期时间
```

### 5.3 缓存雪崩

```
问题：大量 key 同时过期，数据库压力剧增

原因：批量导入数据，同时设置相同 TTL
    TTL=300s 的 10000 个 key
    5分钟后同时过期

解决方案1：随机 TTL
    TTL = base + random(0, 10%)

解决方案2：多级缓存
    L1 本地缓存挡住部分请求

解决方案3：熔断降级
    数据库压力大时，直接拒绝部分请求
```

```python
# 完整缓存解决方案

class CacheService:
    """带防护的缓存服务"""
    
    def __init__(self):
        self.cache = cache
        self.locks = {}  # 简单锁，生产用 Redis 分布式锁
    
    def get_with_lock(self, key: str, loader_func, ttl=300):
        """
        带互斥锁的缓存读取（解决缓存击穿）
        """
        # 1. 查缓存
        value = self.cache.get(key)
        if value is not None:
            return value
        
        # 2. 获取锁（防止缓存击穿）
        if key in self.locks:
            # 等待锁释放
            while key in self.locks:
                time.sleep(0.01)
            return self.cache.get(key)
        
        try:
            self.locks[key] = True
            
            # 双检锁
            value = self.cache.get(key)
            if value is not None:
                return value
            
            # 3. 查数据库
            value = loader_func()
            
            # 4. 写入缓存（空值也缓存，防穿透）
            self.cache.set(key, value, ttl)
            
            return value
        finally:
            del self.locks[key]
    
    def get_with_bloom_filter(self, key: str, loader_func, bloom_filter):
        """
        带布隆过滤器的缓存读取（解决缓存穿透）
        """
        # 1. 布隆过滤器检查
        if not bloom_filter.might_contain(key):
            return None  # 一定不存在
        
        # 2. 正常查缓存
        return self.get_with_lock(key, loader_func)
```

## 七、总结

```
┌─────────────────────────────────────────────────────────────────────────────┐
│                     缓存层总结                                               │
├─────────────────────────────────────────────────────────────────────────────┤
│                                                                              │
│   一句话：在应用和数据库之间加一层高速内存存储，减少数据库压力                │
│                                                                              │
│   核心收益：                                                                 │
│   • 读性能提升 50-100 倍（从 50ms 到 1ms）                                   │
│   • 数据库负载降低 90%+                                                      │
│   • 支持更高并发                                                             │
│                                                                              │
│   实现要点：                                                                 │
│   1. 多级缓存：本地（L1）+ Redis（L2）                                       │
│   2. Cache Aside 策略：先读缓存，miss则查库回填                              │
│   3. 写操作清缓存：保证数据一致性                                            │
│   4. 防三大问题：穿透（缓存空值）、击穿（加锁）、雪崩（随机TTL）              │
│                                                                              │
│   适用数据：                                                                 │
│    读多写少：用户配置、知识库信息                                          │
│    热点数据：热门文档、常用模板                                            │
│    计算昂贵：复杂查询结果                                                  │
│    实时性高：聊天记录、实时状态                                            │
│    数据量大：无法全部放入内存                                              │
│                                                                              │
│   与读写分离的区别：                                                         │
│   • 读写分离：分散数据库压力，但还是查数据库                                  │
│   • 缓存层：避免查数据库，直接内存返回                                        │
│   • 最佳实践：两者结合，缓存 → 从库 → 主库                                   │
│                                                                              │
└─────────────────────────────────────────────────────────────────────────────┘
```
