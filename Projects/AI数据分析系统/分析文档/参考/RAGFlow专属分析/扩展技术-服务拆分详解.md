# 扩展技术：服务拆分（微服务化）详解

**技术类型**: 架构解耦  
**适用场景**: 单体应用过于庞大，需要独立扩展  
**难度**: ⭐⭐⭐ 中高

### 1.2 一句话解释

> **服务拆分 = 把一个大应用拆成多个小服务，每个服务独立开发、部署、扩展**

## 二、为什么需要服务拆分？

### 2.1 单体的问题

```python
# 场景：RAGFlow 的文档上传高峰期

# 用户疯狂上传文档（文档模块压力大）
POST /api/documents      # CPU 使用率 90%
POST /api/documents      # 内存接近上限
POST /api/documents      # 响应变慢 5s+

# 同时其他用户在聊天（Agent模块）
POST /api/chat           # 被文档模块拖累，响应也变慢

# 新用户注册（用户模块）
POST /api/users          # 同样受影响

# 结果：整个系统变慢，全部用户受影响
```

### 2.2 拆分后的改善

```python
# 拆分后：

# 文档服务压力大
POST /api/documents ──► 文档服务 Pod-1  CPU 90%
                        文档服务 Pod-2  CPU 85%
                        文档服务 Pod-3  CPU 88%
                        
# HPA 自动扩容：
文档服务 Pod-4 创建中...
文档服务 Pod-5 创建中...

# 用户服务完全不受影响：
POST /api/users ──► 用户服务 Pod-1  CPU 20%  响应 50ms
POST /api/chat  ──► Agent服务 Pod-1  CPU 30%  响应 200ms

# 结果：只有文档服务需要处理压力，其他服务正常
```

```python
# services/user-service/api/user_api.py

from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel, EmailStr
from typing import List, Optional

from service.user_service import UserService
from common.auth import verify_token, get_current_user

router = APIRouter()


class CreateUserRequest(BaseModel):
    email: EmailStr
    password: str
    nickname: str
    tenant_id: Optional[str] = None


class UserResponse(BaseModel):
    id: str
    email: str
    nickname: str
    tenant_id: str
    avatar: Optional[str] = None
    
    class Config:
        from_attributes = True


@router.post("/", response_model=UserResponse)
async def create_user(
    request: CreateUserRequest,
    current_user: dict = Depends(get_current_user)
):
    """创建用户"""
    # 检查权限
    if not current_user.get('is_admin'):
        raise HTTPException(status_code=403, detail="无权限")
    
    user = await UserService.create_user(
        email=request.email,
        password=request.password,
        nickname=request.nickname,
        tenant_id=request.tenant_id
    )
    
    return user


@router.get("/{user_id}", response_model=UserResponse)
async def get_user(
    user_id: str,
    current_user: dict = Depends(get_current_user)
):
    """获取用户信息"""
    # 只能查自己或同租户的用户
    user = await UserService.get_by_id(user_id)
    
    if not user:
        raise HTTPException(status_code=404, detail="用户不存在")
    
    if user.tenant_id != current_user['tenant_id'] and not current_user.get('is_admin'):
        raise HTTPException(status_code=403, detail="无权限查看此用户")
    
    return user


@router.get("/", response_model=List[UserResponse])
async def list_users(
    tenant_id: Optional[str] = None,
    page: int = 1,
    size: int = 20,
    current_user: dict = Depends(get_current_user)
):
    """查询用户列表"""
    # 如果没指定 tenant_id，使用当前用户的
    if not tenant_id:
        tenant_id = current_user['tenant_id']
    
    # 检查权限
    if tenant_id != current_user['tenant_id'] and not current_user.get('is_admin'):
        raise HTTPException(status_code=403, detail="无权限")
    
    users = await UserService.list_by_tenant(
        tenant_id=tenant_id,
        page=page,
        size=size
    )
    
    return users
```

### 4.3 服务间通信（gRPC）

```protobuf
# services/proto/user.proto

syntax = "proto3";

package user;

option go_package = "github.com/ragflow/proto/user";

// 用户服务
service UserService {
    // 获取用户信息
    rpc GetUser(GetUserRequest) returns (User);
    
    // 验证用户 token
    rpc ValidateToken(ValidateTokenRequest) returns (TokenResponse);
    
    // 获取租户信息
    rpc GetTenant(GetTenantRequest) returns (Tenant);
    
    // 检查权限
    rpc CheckPermission(CheckPermissionRequest) returns (PermissionResponse);
}

message GetUserRequest {
    string user_id = 1;
}

message User {
    string id = 1;
    string email = 2;
    string nickname = 3;
    string tenant_id = 4;
    string avatar = 5;
    repeated string roles = 6;
}

message ValidateTokenRequest {
    string token = 1;
}

message TokenResponse {
    bool valid = 1;
    User user = 2;
    string error = 3;
}

message GetTenantRequest {
    string tenant_id = 1;
}

message Tenant {
    string id = 1;
    string name = 2;
    string plan = 3;  // free, pro, enterprise
}

message CheckPermissionRequest {
    string user_id = 1;
    string resource = 2;
    string action = 3;  // read, write, delete
}

message PermissionResponse {
    bool allowed = 1;
    string reason = 2;
}
```

```python
# services/common/grpc_clients.py

import grpc
import asyncio
from typing import Optional

from proto import user_pb2, user_pb2_grpc


class UserServiceClient:
    """用户服务 gRPC 客户端"""
    
    def __init__(self, target: str = "user-service:8001"):
        self.target = target
        self.channel = None
        self.stub = None
        self._connect()
    
    def _connect(self):
        """建立连接"""
        # 使用连接池
        self.channel = grpc.aio.insecure_channel(
            self.target,
            options=[
                ('grpc.lb_policy_name', 'round_robin'),
                ('grpc.keepalive_time_ms', 10000),
                ('grpc.keepalive_timeout_ms', 5000),
            ]
        )
        self.stub = user_pb2_grpc.UserServiceStub(self.channel)
    
    async def get_user(self, user_id: str) -> Optional[dict]:
        """获取用户信息"""
        try:
            request = user_pb2.GetUserRequest(user_id=user_id)
            response = await self.stub.GetUser(request, timeout=5)
            
            return {
                'id': response.id,
                'email': response.email,
                'nickname': response.nickname,
                'tenant_id': response.tenant_id,
                'avatar': response.avatar,
                'roles': list(response.roles)
            }
        except grpc.RpcError as e:
            print(f"GetUser error: {e}")
            return None
    
    async def validate_token(self, token: str) -> tuple[bool, Optional[dict]]:
        """验证 token"""
        try:
            request = user_pb2.ValidateTokenRequest(token=token)
            response = await self.stub.ValidateToken(request, timeout=3)
            
            if response.valid:
                user = {
                    'id': response.user.id,
                    'email': response.user.email,
                    'tenant_id': response.user.tenant_id,
                }
                return True, user
            else:
                return False, None
        except grpc.RpcError as e:
            print(f"ValidateToken error: {e}")
            return False, None
    
    async def check_permission(
        self, 
        user_id: str, 
        resource: str, 
        action: str
    ) -> bool:
        """检查权限"""
        try:
            request = user_pb2.CheckPermissionRequest(
                user_id=user_id,
                resource=resource,
                action=action
            )
            response = await self.stub.CheckPermission(request, timeout=3)
            return response.allowed
        except grpc.RpcError:
            # 默认拒绝
            return False


# 全局客户端实例（单例）
_user_client = None

async def get_user_client() -> UserServiceClient:
    """获取用户服务客户端（单例）"""
    global _user_client
    if _user_client is None:
        _user_client = UserServiceClient()
    return _user_client
```

### 4.4 知识库服务调用用户服务

```python
# services/kb-service/api/kb_api.py

from fastapi import APIRouter, Depends, HTTPException
from pydantic import BaseModel

from service.kb_service import KnowledgebaseService
from common.grpc_clients import get_user_client

router = APIRouter()


@router.post("/{kb_id}/share")
async def share_knowledgebase(
    kb_id: str,
    target_user_id: str,
    current_user: dict = Depends(get_current_user_from_token)
):
    """
    分享知识库给指定用户
    
    需要调用用户服务验证目标用户是否存在
    """
    # 1. 检查当前用户是否有权限
    kb = await KnowledgebaseService.get_by_id(kb_id)
    if not kb or kb.owner_id != current_user['id']:
        raise HTTPException(status_code=403, detail="无权限")
    
    # 2. 调用用户服务验证目标用户
    user_client = await get_user_client()
    target_user = await user_client.get_user(target_user_id)
    
    if not target_user:
        raise HTTPException(status_code=404, detail="目标用户不存在")
    
    # 3. 检查是否同租户
    if target_user['tenant_id'] != current_user['tenant_id']:
        raise HTTPException(status_code=400, detail="只能分享给同租户用户")
    
    # 4. 执行分享
    result = await KnowledgebaseService.share(kb_id, target_user_id)
    
    return {
        "success": True,
        "shared_with": target_user['email']
    }
```

## 六、优缺点

### 6.1 优点

| 优点 | 说明 |
|------|------|
| **独立扩展** | 热点服务单独扩容，资源精准投入 |
| **故障隔离** | 一个服务挂了不影响其他服务 |
| **技术异构** | 各服务可用最适合的技术栈 |
| **团队自治** | 小团队独立开发、部署、运维 |
| **代码简洁** | 服务内聚，代码量少，易于理解 |

### 6.2 缺点与挑战

| 缺点 | 解决方案 |
|------|----------|
| **分布式事务** | 使用 Saga 模式，最终一致性 |
| **服务间调用复杂** | 使用 Service Mesh（Istio）简化 |
| **网络延迟** | 服务内聚，减少不必要调用 |
| **运维复杂** | 使用 K8s + DevOps 自动化 |
| **测试困难** | 契约测试、Mock 服务 |
