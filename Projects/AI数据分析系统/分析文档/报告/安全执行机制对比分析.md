**核心实现**:
```python
# TaskWeaver代码执行引擎
class CodeExecutor:
    def __init__(self, use_container=True):
        self.use_container = use_container
        self.container_config = {
            "image": "taskweaver-executor:latest",
            "cpu_limit": "1.0",      # 限制CPU
            "memory_limit": "512m",   # 限制内存
            "network_mode": "none",   # 禁止网络
            "volumes": {              # 只读挂载
                "/data": {"bind": "/data", "mode": "ro"}
            }
        }
    
    def execute(self, code, timeout=60):
        if self.use_container:
            return self._execute_in_container(code, timeout)
        else:
            return self._execute_in_subprocess(code, timeout)
    
    def _execute_in_container(self, code, timeout):
        import docker
        client = docker.from_env()
        
        # 创建临时文件
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            f.write(code)
            temp_file = f.name
        
        try:
            # 在Docker容器中执行
            container = client.containers.run(
                **self.container_config,
                command=f"python {temp_file}",
                detach=True,
                auto_remove=True
            )
            
            # 等待结果，带超时
            result = container.wait(timeout=timeout)
            logs = container.logs().decode('utf-8')
            
            return {"output": logs, "exit_code": result['StatusCode']}
        except docker.errors.ContainerError as e:
            return {"error": str(e), "exit_code": 1}
```

**安全特性**:
| 特性 | 实现方式 | 安全等级 |
|------|---------|---------|
| **资源限制** | Docker cgroup (CPU/内存) | ⭐⭐⭐⭐⭐ |
| **网络隔离** | `--network none` | ⭐⭐⭐⭐⭐ |
| **文件隔离** | 只读挂载+临时卷 | ⭐⭐⭐⭐⭐ |
| **超时控制** | `container.wait(timeout=60)` | ⭐⭐⭐⭐⭐ |
| **自动清理** | `auto_remove=True` | ⭐⭐⭐⭐⭐ |

**借鉴价值**: **极高** - Docker容器提供最完整的隔离，适合生产环境

```python
# premsql执行器
class Executor:
    def execute(self, sql, connection):
        """
        执行SQL并返回结果
        """
        try:
            # 1. SQL语法验证
            self._validate_sql(sql)
            
            # 2. 权限检查
            if not self._check_permissions(sql, connection):
                raise PermissionError("SQL operation not permitted")
            
            # 3. 在隔离连接中执行
            result = connection.execute(sql)
            return result
        except Exception as e:
            return {"error": str(e)}
    
    def _validate_sql(self, sql):
        """
        验证SQL语法，阻止危险操作
        """
        dangerous_keywords = [
            'DROP', 'DELETE', 'TRUNCATE', 
            'ALTER', 'GRANT', 'REVOKE'
        ]
        
        sql_upper = sql.upper()
        for keyword in dangerous_keywords:
            if keyword in sql_upper:
                raise ValueError(f"Dangerous SQL operation detected: {keyword}")
```

- 只读模式（可选）
- 权限检查

```python
class OpenInterpreter:
    def __init__(self):
        self.safe_mode = True  # 安全模式默认开启

        # 1. LLM生成代码
        code = self.llm.generate(message)
        
        # 2. 安全确认
        if self.safe_mode:
            print("\n" + "="*50)
            print("即将执行以下代码：")
            print("="*50)
            print(code)
            print("="*50)
```

### 3.1 数据权限控制

#### Vanna 2.0 ⭐⭐⭐⭐⭐

**权限架构**:
```
用户请求
    ↓
UserResolver (解析用户身份)
    ↓
User (用户对象，携带权限信息)
    ↓
ToolContext (工具上下文，注入权限)
    ↓
权限检查 → SQL生成/执行
```

**核心实现**:
```python
class UserAwareArchitecture:
    """
    Vanna用户感知架构
    """
    def __init__(self):
        self.user_resolver = UserResolver()
        self.permission_manager = PermissionManager()
    
    def execute_query(self, user_token, query):
        # 1. 解析用户
        user = self.user_resolver.resolve(user_token)
        
        # 2. 检查权限
        if not self.permission_manager.can_query(user, query):
            raise PermissionDenied("User does not have permission")
        
        # 3. 注入行级安全
        secure_query = self._apply_row_level_security(user, query)
        
        # 4. 执行
        return self.database.execute(secure_query)
    
    def _apply_row_level_security(self, user, query):
        """
        应用行级安全策略
        """
        # 示例：用户只能看到自己的数据
        rls_clause = f"tenant_id = '{user.tenant_id}'"
        
        # 将RLS条件注入WHERE子句
        modified_query = inject_where_clause(query, rls_clause)
        return modified_query
```

**权限粒度**:
| 级别 | 控制内容 | 实现方式 |

| **表级** | 允许查询的表 | 权限矩阵 |
| **行级** | 允许访问的数据行 | RLS策略 |
| **列级** | 允许访问的字段 | 视图/屏蔽 |

```python
class DockerCodeExecutor:
    """
    基于Docker的代码执行器
    """
    def __init__(self, 
                 image="python:3.11-slim",
                 cpu_limit="1.0",
                 memory_limit="512m",
                 timeout=60):
        self.image = image
        self.cpu_limit = cpu_limit
        self.memory_limit = memory_limit
        self.timeout = timeout
        self.client = docker.from_env()
    
    def execute(self, code: str) -> dict:
        # 创建临时文件
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            f.write(code)
            temp_file = f.name
        
        try:
            # 运行容器
            container = self.client.containers.run(
                self.image,
                f"python {temp_file}",
                cpu_period=100000,
                cpu_quota=int(float(self.cpu_limit) * 100000),
                mem_limit=self.memory_limit,
                network_mode="none",
                read_only=True,
                detach=True
            )
            
            # 等待结果
            result = container.wait(timeout=self.timeout)
            logs = container.logs().decode('utf-8')
            
            return {
                "output": logs,
                "exit_code": result['StatusCode'],
                "success": result['StatusCode'] == 0
            }
        except Exception as e:
            return {"error": str(e), "success": False}
        finally:
            # 清理
            container.remove(force=True)
```

```python
class SubprocessSandbox:
    """
    基于子进程的轻量级沙箱
    """
    def execute(self, code: str, timeout: int = 60) -> dict:
        import resource
        
        def set_limits():
            # 设置资源限制
            resource.setrlimit(resource.RLIMIT_CPU, (timeout, timeout))
            resource.setrlimit(resource.RLIMIT_AS, (512*1024*1024, 512*1024*1024))
        
        with tempfile.NamedTemporaryFile(mode='w', suffix='.py', delete=False) as f:
            f.write(code)
            temp_file = f.name
        
        try:
            result = subprocess.run(
                ['python', temp_file],
                capture_output=True,
                text=True,
                timeout=timeout,
                preexec_fn=set_limits  # Linux/Mac
            )
            return {
                "output": result.stdout,
                "error": result.stderr,
                "success": result.returncode == 0
            }
        except subprocess.TimeoutExpired:
            return {"error": "Execution timeout", "success": False}
        finally:
            os.unlink(temp_file)
```

```python
class SQLValidator:

    SQL安全验证器
    """

        'DROP', 'DELETE', 'TRUNCATE', 'ALTER',
        'GRANT', 'REVOKE', 'CREATE', 'INSERT', 'UPDATE'
    ]
    
    def __init__(self, read_only=True):
        self.read_only = read_only
    
    def validate(self, sql: str) -> tuple[bool, str]:
        sql_upper = sql.upper()
        
        # 检查危险关键词
        for keyword in self.DANGEROUS_KEYWORDS:
            if keyword in sql_upper:
                if self.read_only:
                    return False, f"{keyword} is not allowed in read-only mode"
        
        return True, "SQL is valid"
```

## 七、总结与建议

### 7.1 核心结论

1. **Docker容器提供最完整的隔离**（TaskWeaver）
2. **子进程沙箱是轻量级选择**（AI Data Science Team, PandasAI）
3. **人工确认是重要的安全补充**（Open Interpreter）
4. **权限控制是生产环境必备**（Vanna, SuperSonic）
