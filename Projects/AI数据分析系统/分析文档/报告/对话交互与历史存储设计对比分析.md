```python
class PandasAIAgent:
    def __init__(self):
        self.memory = ConversationBufferMemory()
        self.context = {
            "dataframes": {},  # 多个DataFrame上下文
            "last_query": None,
            "last_result": None
        }
    
    def chat(self, query: str) -> str:
        # 1. 解析指代消解
        resolved_query = self._resolve_references(query)
        # 2. 构建带上下文的Prompt
        prompt = self._build_prompt(resolved_query)
        # 3. 生成代码
        code = self.llm.generate(prompt)
        # 4. 执行并返回
        result = self._execute(code)
        # 5. 更新记忆
        self.memory.add(query, result)
        return result
```

```python
class OpenInterpreter:
    def __init__(self):
        # 简单的消息列表存储
        self.messages = []  # 内存存储，进程结束即丢失
    
    def chat(self, message):
        self.messages.append({"role": "user", "content": message})
        response = self.llm.generate(self.messages)
        self.messages.append({"role": "assistant", "content": response})
        return response
    
    def export(self, file_path):
        """导出对话历史到文件"""
        import json
        with open(file_path, 'w') as f:
            json.dump(self.messages, f)
    
    def load(self, file_path):
        """从文件加载对话历史"""
        import json
        with open(file_path, 'r') as f:
            self.messages = json.load(f)
```

**特点**: 简单直接，无持久化，需手动导出

#### TaskWeaver - Round/Post结构化存储 ⭐⭐⭐⭐

```python
class TaskWeaverMemory:
    """
    TaskWeaver的记忆系统设计
    """
    def __init__(self):
        self.conversation = Conversation()  # 对话历史
        self.shared_memory = SharedMemory()  # 共享状态
        self.experience = Experience()  # 经验学习
    
    class Conversation:
        def __init__(self):
            self.rounds = []  # 多轮对话列表
        
        def add_round(self, user_query, planner_plan, code_result):
            """
            添加一轮完整的对话记录
            """
            round_record = {
                "round_id": len(self.rounds),
                "timestamp": datetime.now(),
                "user_query": user_query,
                "planner_plan": planner_plan,  # Planner的规划
                "code_interpreter_posts": code_result,  # CodeInterpreter的执行结果
                "shared_state_snapshot": self._capture_state()
            }
            self.rounds.append(round_record)
    
    class SharedMemory:
        """
        共享内存 - 保存执行过程中的变量
        """
        def __init__(self):
            self.variables = {}  # 变量存储
            self.dataframes = {}  # DataFrame存储
        
        def set(self, name, value):
            self.variables[name] = value
        
        def get(self, name):
            return self.variables.get(name)
    
    class Experience:
        """
        经验学习 - 从历史对话中提取成功模式
        """
        def learn_from_history(self, successful_rounds):
            """
            分析成功的对话，提取可复用的经验
            """
            patterns = self._extract_patterns(successful_rounds)
            self._store_experience(patterns)
```

**特点**:
-  结构化记录（Round/Post）
-  状态快照（可回溯）
-  经验学习（从成功中提取模式）
-  共享内存（变量保持）

#### Vanna 2.0 - 多层级持久化存储 ⭐⭐⭐⭐⭐

```python
class VannaStorage:
    """
    Vanna的多层级存储设计
    """
    def __init__(self):
        self.storage_backend = self._init_storage()
    
    def _init_storage(self):
        # 支持多种存储后端
        return {
            "sqlite": SQLiteStorage(),
            "postgres": PostgresStorage(),
            "s3": S3Storage()
        }[config.STORAGE_TYPE]
    
    class ConversationStorage:
        """
        对话历史存储
        """
        def save_conversation(self, user_id, conversation_id, messages):
            """
            存储完整对话
            """
            data = {
                "user_id": user_id,
                "conversation_id": conversation_id,
                "messages": messages,  # 结构化消息列表
                "created_at": datetime.now(),
                "updated_at": datetime.now()
            }
            self.storage_backend.insert("conversations", data)
        
        def get_conversation_history(self, user_id, limit=50):
            """
            获取用户的对话历史
            """
            return self.storage_backend.query(
                "SELECT * FROM conversations WHERE user_id = ? ORDER BY updated_at DESC LIMIT ?",
                (user_id, limit)
            )
    
    class VectorStorage:
        """
        向量存储 - 用于语义检索
        """
        def __init__(self):
            self.vector_store = ChromaDB()  # 或Pinecone/Faiss
        
        def store_question_sql_pair(self, question, sql, metadata):
            """
            存储问题-SQL对，用于RAG检索
            """
            embedding = self.embedding_model.encode(question)
            self.vector_store.add(
                ids=[hash(question)],
                embeddings=[embedding],
                metadatas=[{
                    "question": question,
                    "sql": sql,
                    **metadata
                }]
            )
        
        def retrieve_similar_questions(self, query, k=5):
            """
            检索相似的历史问题
            """
            query_embedding = self.embedding_model.encode(query)
            results = self.vector_store.similarity_search(query_embedding, k=k)
            return results
```

**特点**:
-  多后端支持（SQLite/PostgreSQL/S3）
-  向量化存储（语义检索）
-  用户隔离（user_id分区）
-  完整的审计日志

```python
class SuperSonicChatMemory:
    """
    SuperSonic的对话记忆管理
    """
    def __init__(self):
        self.short_term_memory = []  # 短期记忆（当前会话）
        self.long_term_memory = {}   # 长期记忆（用户偏好）
    
    def add_message(self, role, content, message_type="text"):
        """
        添加消息到短期记忆
        """
        message = {
            "role": role,  # user/assistant/system
            "content": content,
            "type": message_type,  # text/sql/chart
            "timestamp": datetime.now(),
            "context": self._extract_context(content)  # 提取关键上下文
        }
        self.short_term_memory.append(message)
        
        # 保持窗口大小
        if len(self.short_term_memory) > 20:
            self.short_term_memory.pop(0)
    
    def get_context_for_llm(self, window_size=10):
        """
        获取用于LLM的上下文窗口
        """
        return self.short_term_memory[-window_size:]
    
    def save_user_preference(self, user_id, preference):
        """
        保存用户长期偏好
        """
        self.long_term_memory[user_id] = {
            "preferred_charts": preference.get("charts", []),
            "common_tables": preference.get("tables", []),
            "query_patterns": preference.get("patterns", [])
        }
```

**特点**:
-  短/长期记忆分离
-  窗口管理（防止超长）
-  上下文提取（关键信息）
-  用户偏好学习

| 消息类型 | 用途 | 代表项目 |

### 4.2 上下文管理策略

| 策略 | 说明 | 代表项目 |
|------|------|---------|

| **Token限制** | 按Token数截断 | Vanna |
| **摘要压缩** | 早期对话生成摘要 | 理论上可用 |
| **RAG检索** | 向量检索相关历史 | Vanna, chatDuck |

```python
class ConversationStorage:
    """
    推荐的对话存储设计
    """
    def __init__(self):
        # 主存储 - PostgreSQL/SQLite
        self.db = SQLAlchemyEngine()
        # 向量存储 - 语义检索
        self.vector_store = ChromaDB()
    
    def save_message(self, conversation_id, user_id, message):
        """
        保存消息到数据库
        """
        # 1. 保存到关系数据库
        self.db.insert("messages", {
            "conversation_id": conversation_id,
            "user_id": user_id,
            "role": message["role"],
            "content": message["content"],
            "message_type": message.get("type", "text"),
            "metadata": json.dumps(message.get("metadata", {})),
            "created_at": datetime.now()
        })
        
        # 2. 如果是用户问题，添加到向量存储
        if message["role"] == "user":
            embedding = self.get_embedding(message["content"])
            self.vector_store.add(
                ids=[f"{conversation_id}_{message['id']}"],
                embeddings=[embedding],
                metadatas=[{
                    "conversation_id": conversation_id,
                    "user_id": user_id,
                    "question": message["content"],
                    "sql": message.get("sql", ""),
                    "timestamp": datetime.now()
                }]
            )
    
    def get_conversation_context(self, conversation_id, window_size=10):
        """
        获取对话上下文
        """
        # 获取最近N条消息
        recent_messages = self.db.query(
            "SELECT * FROM messages WHERE conversation_id = ? ORDER BY created_at DESC LIMIT ?",
            (conversation_id, window_size)
        )
        return reversed(recent_messages)  # 按时间正序
    
    def retrieve_similar_history(self, user_id, query, k=3):
        """
        检索用户的历史相似问题
        """
        query_embedding = self.get_embedding(query)
        results = self.vector_store.similarity_search(
            query_embedding, 
            filter={"user_id": user_id},
            k=k
        )
        return results
```

### 6.2 上下文管理器

```python
class ConversationManager:
    """
    对话上下文管理器
    """
    def __init__(self, max_window_size=10, max_tokens=4000):
        self.max_window_size = max_window_size
        self.max_tokens = max_tokens
        self.storage = ConversationStorage()
    
    def add_message(self, conversation_id: str, message: Message):
        """添加消息"""
        self.storage.save_message(conversation_id, message)
    
    def get_context(self, conversation_id: str) -> List[Message]:
        """
        获取对话上下文（智能截断）
        """
        messages = self.storage.get_recent_messages(
            conversation_id, 
            limit=self.max_window_size
        )
        
        # Token数检查，如果超限则截断
        total_tokens = sum(self._estimate_tokens(m.content) for m in messages)
        while total_tokens > self.max_tokens and len(messages) > 1:
            # 移除最早的消息（保留system消息）
            if messages[0].role != "system":
                removed = messages.pop(0)
                total_tokens -= self._estimate_tokens(removed.content)
            else:
                break
        
        return messages
    
    def _estimate_tokens(self, text: str) -> int:
        """估算Token数（简化版）"""
        return len(text) // 4  # 粗略估计
```
