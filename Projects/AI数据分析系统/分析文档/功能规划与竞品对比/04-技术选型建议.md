### 3.3 MCP Client 实现建议

```python
# app/agents/mcp/client.py
from mcp import ClientSession, StdioServerParameters
from mcp.client.stdio import stdio_client

class MCPToolManager:
    """MCP 工具管理器"""
    
    def __init__(self):
        self.servers: Dict[str, ClientSession] = {}
        self.tools: Dict[str, Tool] = {}
    
    async def connect_server(self, name: str, config: ServerConfig):
        """连接 MCP Server"""
        params = StdioServerParameters(
            command=config.command,
            args=config.args,
            env=config.env
        )
        
        async with stdio_client(params) as (read, write):
            async with ClientSession(read, write) as session:
                await session.initialize()
                self.servers[name] = session
                
                # 发现工具
                tools = await session.list_tools()
                for tool in tools.tools:
                    self.tools[f"{name}__{tool.name}"] = self._adapt_tool(
                        session, tool
                    )
    
    def _adapt_tool(self, session: ClientSession, mcp_tool) -> Tool:
        """适配为 smolagents Tool"""
        
        @tool
        async def wrapper(**kwargs):
            result = await session.call_tool(
                mcp_tool.name,
                arguments=kwargs
            )
            return result.content[0].text
        
        wrapper.name = f"mcp_{mcp_tool.name}"
        wrapper.description = mcp_tool.description
        return wrapper
```

## 五、流式输出优化

### 5.1 当前架构评估

| 方面 | 现状 | 问题 | 优化方向 |
|------|------|------|----------|
| 传输协议 | SSE | 单向 | 考虑 WebSocket |
| 事件格式 | 自定义 JSON | 无标准 | 标准化事件 |

### 5.2 优化建议

**事件标准化**:
```typescript
// 统一事件格式
interface AgentEvent {
  id: string;           // 事件唯一 ID
  type: EventType;      // 事件类型
  timestamp: number;    // 时间戳
  sequence: number;     // 序列号（用于重排/恢复）
  source: string;       // 来源 (host/worker/tool)
  data: EventData;      // 事件数据
}

type EventType = 
  | "thought"      // 思考过程
  | "tool_call"    // 工具调用
  | "tool_result"  // 工具结果
  | "code"         // 代码生成
  | "output"       // 输出
  | "error"        // 错误
  | "progress"     // 进度
  | "final_answer"; // 最终答案
```

### 6.2 Docker Sandbox 方案

```python
# 基于已有设计
class DockerExecutor:
    """Docker 隔离执行环境"""
    
    def __init__(self, image: str = "aiasys-python-env"):
        self.client = docker.from_env()
        self.image = image
    
    async def execute(self, code: str, timeout: int = 300) -> ExecutionResult:
        container = self.client.containers.run(
            self.image,
            command=["python", "-c", code],
            mem_limit="512m",
            cpu_quota=100000,  # 1 CPU
            network_mode="none",  # 禁用网络
            detach=True,
            remove=True
        )
        
        try:
            result = container.wait(timeout=timeout)
            logs = container.logs().decode()
            return ExecutionResult(
                success=result["StatusCode"] == 0,
                output=logs
            )
        except:
            container.kill()
            raise
```

```python
# 根据任务选择模型
class ModelRouter:
    """智能模型路由"""
    
    def select_model(self, task: Task) -> str:
        """根据任务特征选择最优模型"""
        
        if task.complexity == "high":
            return "gpt-4o"  # 复杂任务用强模型
        elif task.type == "code":
            return "claude-3-sonnet"  # 代码任务
        elif task.require_streaming:
            return "gpt-4o-mini"  # 流式用快模型
        else:
            return "gpt-4o-mini"  # 默认用经济模型
```

**故障转移**:
```python
# 多供应商备份
class FallbackLLM:
    def __init__(self):
        self.providers = [
            OpenAIProvider(),
            AnthropicProvider(),
            AzureProvider()
        ]
    
    async def generate(self, prompt: str):
        for provider in self.providers:
            try:
                return await provider.generate(prompt)
            except Exception as e:
                logger.warning(f"{provider.name} failed: {e}")
                continue
        raise AllProvidersFailed()
```
