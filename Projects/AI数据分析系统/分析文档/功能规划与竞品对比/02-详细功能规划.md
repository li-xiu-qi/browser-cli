**集成方案**:
```python
# MCP Client 集成
from mcp import ClientSession

class MCPToolAdapter:
    """将 MCP Server 工具适配为 smolagents Tool"""
    
    def __init__(self, session: ClientSession):
        self.session = session
        self.tools = []
    
    async def discover_tools(self):
        """发现可用工具"""
        response = await self.session.list_tools()
        for tool in response.tools:
            self.tools.append(self._adapt_tool(tool))
    
    def _adapt_tool(self, mcp_tool) -> Tool:
        """转换为 smolagents Tool"""
        @tool
        async def wrapper(**kwargs):
            result = await self.session.call_tool(
                mcp_tool.name, 
                arguments=kwargs
            )
            return result.content
        
        wrapper.name = mcp_tool.name
        wrapper.description = mcp_tool.description
        return wrapper
```

workflow:
  name: "设备故障诊断"
  
  steps:
    - name: "加载数据"
      type: load
      source: "file"
      path: "{{ input.file_path }}"
      
    - name: "频谱分析"
      type: analyze
      method: "fft"
      input: "{{ steps.加载数据.output }}"
      
    - name: "故障诊断"
      type: llm
      model: "gpt-4o"
      prompt: |
        根据以下频谱分析结果判断故障类型:
